using System; namespace aufgabe1-1 { class Program { static void Main(string[] args) { switch (args[0]){ Console.WriteLine("w fuer Wuerfel, k fuer Kugel, o fuer Oktaeder"); Console.WriteLine("Gebe w, k oder o ein"); case "w": getCubeInfo(double.Parse(args[1])); break; case "k": getSphereInfo(double.Parse(args[1])) break; case "o": getOctaederInfo(double.Parse(args[1])); break; } //get surface cube static double getCubeA (double d) { double A = 6 * d * d; return(A); } //get surface sphere static double getSphereA (double d) { double A = 3.14 * d * d; return(A); } //get surface oktaeder static double getOktaederA (double d) { double A = 2 * Math.Sqrt(3) * d * d; return(A); } //get volume cube static double getCubeV (double d) { double V = d * d * d; return(V); } //get volume sphere static double getSphereV (double d) { double V = (3.14 * d * d * d) / 6; return(V); } //get volume oktaeder static double getOktaederV (double d) { double V = (Math.Sqrt(2) * d * d * d) / 3; return(V); } } }
